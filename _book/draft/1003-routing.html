3.2. 라우팅 만들기

3.2.1. 웰컴 뷰 수정

<!-- //코드 3-1 resources/views/welcome.blade.php -->

<h1>Hello Laravel</h1>

# 콘솔 3-1 로컬 서버 구동

$ php artisan serve

그림 3-1 웰컴 뷰 조작

3.2.2. 라우팅이란 ?

// 코드 3-2 app/Http/routes.php

Route::get('/', function () {
    return view('welcome');
});

3.2.3. 라우팅과 HTTP 응답

// 코드 3-3 app/Http/routes.php

Route::get('/', function () {
    return '<h1>Hello Foo</h1>';
});

3.3. URL 파라미터

// 코드 3-4 app/Http/routes.php

Route::get('/{foo}', function ($foo) {
    return $foo;
});

// 코드 3-5 app/Http/routes.php

Route::get('/{foo?}', function ($foo = 'bar') {
    return $foo;
});

3.3.1. URL 파라미터 패턴

// 코드 3-6 app/Http/routes.php

Route::pattern('foo', '[0-9a-zA-Z]{3}');

Route::get('/{foo?}', function ($foo = 'bar') {
    return $foo;
});

// 코드 3-7 app/Http/routes.php

Route::get('/{foo?}', function ($foo = 'bar') {
    return $foo;
})->where('foo', '[0-9a-zA-Z]{3}');

3.4. 라우트 이름

// 코드 3-8 app/Http/routes.php

Route::get('/', [
    'as' => 'home',
    function () {
        return '제 이름은 "home" 입니다.';
    }
]);

Route::get('/home', function () {
    return redirect(route('home'));
});
